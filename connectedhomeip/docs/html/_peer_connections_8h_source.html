299  template <typename Callback>
300  void ExpireInactiveConnections(uint64_t maxIdleTimeMs, Callback callback)
301  {
302  const uint64_t currentTime = mTimeSource.GetCurrentMonotonicTimeMs();
303 
304  for (size_t i = 0; i < kMaxConnectionCount; i++)
305  {
306  if (!mStates[i].GetPeerAddress().IsInitialized())
307  {
308  continue; // not an active connection
309  }
310 
311  uint64_t connectionActiveTime = mStates[i].GetLastActivityTimeMs();
312  if (connectionActiveTime + maxIdleTimeMs >= currentTime)
313  {
314  continue; // not expired
315  }
316 
317  MarkConnectionExpired(&mStates[i], callback);
318  }
319  }
