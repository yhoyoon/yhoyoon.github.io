227 template <class T, unsigned int N>
228 inline T * ObjectPool<T, N>::TryCreate(Layer & aLayer)
229 {
230  T * lReturn = nullptr;
231  unsigned int lIndex;
232 #if CHIP_SYSTEM_CONFIG_PROVIDE_STATISTICS
233  unsigned int lNumInUse = 0;
234 #endif
235 
236  for (lIndex = 0; lIndex < N; ++lIndex)
237  {
238  T & lObject = reinterpret_cast<T *>(mArena.uMemory)[lIndex];
239 
240  if (lObject.TryCreate(aLayer, sizeof(T)))
241  {
242  lReturn = &lObject;
243  break;
244  }
245  }
246 
247 #if CHIP_SYSTEM_CONFIG_PROVIDE_STATISTICS
248  if (lReturn != nullptr)
249  {
250  lIndex++;
251  lNumInUse = lIndex;
252  GetNumObjectsInUse(lIndex, lNumInUse);
253  }
254  else
255  {
256  lNumInUse = N;
257  }
258 
259  UpdateHighWatermark(lNumInUse);
260 #endif
261 
262  return lReturn;
263 }
