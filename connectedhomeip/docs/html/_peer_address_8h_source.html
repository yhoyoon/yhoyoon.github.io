123  void ToString(char * buf, size_t bufSize) const
124  {
125  char ip_addr[kInetMaxAddrLen];
126 
127  switch (mTransportType)
128  {
129  case Type::kUndefined:
130  snprintf(buf, bufSize, "UNDEFINED");
131  break;
132  case Type::kUdp:
133  mIPAddress.ToString(ip_addr, sizeof(ip_addr));
134  snprintf(buf, bufSize, "UDP:%s:%d", ip_addr, mPort);
135  break;
136  case Type::kTcp:
137  mIPAddress.ToString(ip_addr, sizeof(ip_addr));
138  snprintf(buf, bufSize, "TCP:%s:%d", ip_addr, mPort);
139  break;
140  case Type::kBle:
141  // Note that BLE does not currently use any specific address.
142  snprintf(buf, bufSize, "BLE");
143  break;
144  default:
145  snprintf(buf, bufSize, "ERROR");
146  break;
147  }
148  }
