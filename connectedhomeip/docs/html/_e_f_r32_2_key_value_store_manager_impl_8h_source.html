93  pw::Status Erase(Address flash_address, size_t num_sectors) override
94  {
95  assert((flash_address % sizeof(uint32_t)) == 0);
96  for (size_t i = 0; i < num_sectors; i++)
97  {
98  auto status =
99  MscStatusToPwStatus(MSC_ErasePage(reinterpret_cast<uint32_t *>(flash_address + i * sector_size_bytes())));

104  }

117  return pw::StatusWithSize(MscStatusToPwStatus(MSC_WriteWord(reinterpret_cast<uint32_t *>(destination_flash_address),

123  static pw::Status MscStatusToPwStatus(MSC_Status_TypeDef msc_status)
124  {
125  switch (msc_status)

128  return pw::OkStatus();
129  case mscReturnUnaligned:
