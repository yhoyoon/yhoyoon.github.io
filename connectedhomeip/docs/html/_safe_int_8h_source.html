37 template <typename T, typename U>
38 bool CanCastTo(U arg)
39 {
40  using namespace std;
41  // U might be a reference to an integer type, if we're assigning from
42  // something passed by reference.
43  typedef typename remove_reference<U>::type V; // V for "value"
44  static_assert(is_integral<T>::value, "Must be assigning to an integral type");
45  static_assert(is_integral<V>::value, "Must be assigning from an integral type");
46 
47  // We want to check that "arg" can fit inside T but without doing any tests
48  // that are always true or always false due to the types involved, which
49  // would trigger compiler warnings themselves. So for example, we can't
50  // compare arg to max values for T if all U values are representable in T,
51  // etc, because those trigger warnings on some compilers.
52 
53  // We also can't directly compare signed to unsigned values in general,
54  // because that will trigger sign conversion warnings. In fact, it will
55  // trigger them even on runtime-unreached codepaths, so for example we can't
56  // directly compare two min() values to each other!
57 
58  // Oh, and some compilers warn on theoretical signed-to-unsigned compares
59  // even when those can't be reached, and that's known at compile time.
60  // Hence all the casts to intmax_t and uintmax_t below.
61 
62  // A bunch of these tests could sure benefit from "if constexpr", but let's
63  // hope compilers just manage to optimize them properly anyway.
64  // We can't blindly compare "arg" to the minimal or maximal value of T one
65  // of T and V is signed and the other is unsigned: there might not be a
66  // single integer type that can represent _both_ the value of arg and the
67  // minimal/maximal value.
68  if (numeric_limits<T>::is_signed && numeric_limits<V>::is_signed)
69  {
70  if (static_cast<intmax_t>(numeric_limits<V>::max()) <= static_cast<intmax_t>(numeric_limits<T>::max()) &&
71  static_cast<intmax_t>(numeric_limits<V>::min()) >= static_cast<intmax_t>(numeric_limits<T>::min()))
72  {
73  // Any checks on arg would be trivially true; don't even do them, to
74  // avoid warnings.
75  return true;
76  }
77 
78  return static_cast<intmax_t>(numeric_limits<T>::min()) <= static_cast<intmax_t>(arg) &&
79  static_cast<intmax_t>(arg) <= static_cast<intmax_t>(numeric_limits<T>::max());
80  }
81 
82  if (!numeric_limits<T>::is_signed && !numeric_limits<V>::is_signed)
83  {
84  if (static_cast<uintmax_t>(numeric_limits<V>::max()) <= static_cast<uintmax_t>(numeric_limits<T>::max()))
85  {
86  // Any checks on arg would be trivially true; don't even do them, to
87  // avoid warnings.
88  return true;
89  }
90 
91  return static_cast<uintmax_t>(arg) <= static_cast<uintmax_t>(numeric_limits<T>::max());
92  }
93 
94  if (numeric_limits<T>::is_signed)
95  {
96  static_assert(numeric_limits<T>::max() >= 0, "What weird type is this?");
97  if (static_cast<uintmax_t>(numeric_limits<V>::max()) <= static_cast<uintmax_t>(numeric_limits<T>::max()))
98  {
99  return true;
100  }
101 
102  return static_cast<uintmax_t>(arg) <= static_cast<uintmax_t>(numeric_limits<T>::max());
103  }
104 
105  return 0 <= arg && static_cast<uintmax_t>(arg) <= static_cast<uintmax_t>(numeric_limits<T>::max());
106 }

122 template <typename T>
123 typename std::enable_if<std::is_unsigned<T>::value, typename std::make_signed<T>::type>::type CastToSigned(T arg)
124 {
125  using namespace std;
126  typedef typename make_signed<T>::type signed_type;
127 
128  if (arg <= static_cast<T>(numeric_limits<signed_type>::max()))
129  {
130  return static_cast<signed_type>(arg);
131  }
132 
133  // We want to return arg - (numeric_limits<T>::max() + 1), but do it without
134  // hitting overflow. We do this by rewriting it as:
135  //
136  // -(numeric_limits<T>::max() - arg) - 1
137  //
138  // then noting that both (numeric_limits<T>::max() - arg) and its negation
139  // are guaranteed to fit in signed_type.
140  signed_type diff = static_cast<signed_type>(numeric_limits<T>::max() - arg);
141  return static_cast<signed_type>(-diff - 1);
142 }
