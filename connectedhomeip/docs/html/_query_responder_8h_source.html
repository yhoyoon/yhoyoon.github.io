132  bool Accept(Internal::QueryResponderInfo * record) const
133  {
134  if (record->responder == nullptr)
135  {
136  return false;
137  }
138 
139  if (mIncludeAdditionalRepliesOnly && !record->reportNowAsAdditional)
140  {
141  return false;
142  }
143 
144  if ((mIncludeOnlyMulticastBeforeMS > 0) && (record->lastMulticastTime >= mIncludeOnlyMulticastBeforeMS))
145  {
146  return false;
147  }
148 
149  if ((mReplyFilter != nullptr) &&
150  !mReplyFilter->Accept(record->responder->GetQType(), record->responder->GetQClass(), record->responder->GetQName()))
151  {
152  return false;
153  }
154  return true;
155  }
156 
157 private:
158  bool mIncludeAdditionalRepliesOnly = false;
159  ReplyFilter * mReplyFilter = nullptr;
160  uint64_t mIncludeOnlyMulticastBeforeMS = 0;
161 };

204 private:
207  void SkipInvalid()
208  {
209  while ((mRemaining > 0) && !mFilter->Accept(mCurrent))
210  {
211  mRemaining--;

217  }
218  }
