49  }

54  mEnableIPv4 = enable;

56  }
57  bool IsIPv4Enabled() const { return mEnableIPv4; }
58  Derived & SetMac(chip::ByteSpan mac)
59  {
60  mMac = chip::ByteSpan(mMacStorage, std::min(mac.size(), kMaxMacSize));
61  memcpy(mMacStorage, mac.data(), mMac.size());

63  }
64  const chip::ByteSpan GetMac() const { return mMac; }
65 
66 private:
67  uint16_t mPort = CHIP_PORT;
68  bool mEnableIPv4 = true;
69  uint8_t mMacStorage[kMaxMacSize] = {};
